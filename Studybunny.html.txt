// Synth & Audio Engine Elements
let audioCtx = null;
let noiseNode = null;
let filterNode = null;

function initAudio() {
    if (!audioCtx) {
        audioCtx = new (window.AudioContext || window.webkitAudioContext)();
    }
}

function playChime() {
    try {
        initAudio();
        const playTone = (freq, startTime, duration) => {
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.type = "sine";
            osc.frequency.setValueAtTime(freq, startTime);
            gain.gain.setValueAtTime(0.2, startTime);
            gain.gain.exponentialRampToValueAtTime(0.001, startTime + duration);
            osc.connect(gain);
            gain.connect(audioCtx.destination);
            osc.start(startTime);
            osc.stop(startTime + duration);
        };
        const now = audioCtx.currentTime;
        playTone(523.25, now, 0.15); 
        playTone(659.25, now + 0.1, 0.15); 
        playTone(783.99, now + 0.2, 0.3); 
    } catch(e) { console.log(e); }
}

function generateNoiseBuffer(type) {
    const bufferSize = 2 * audioCtx.sampleRate;
    const noiseBuffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
    const output = noiseBuffer.getChannelData(0);
    let lastOut = 0.0; // For cafe/rain filtering modeling
    
    for (let i = 0; i < bufferSize; i++) {
        const white = Math.random() * 2 - 1;
        if (type === 'white') {
            output[i] = white * 0.4;
        } else if (type === 'rain') {
            // Pink-brownish heavy crackle
            lastOut = (lastOut * 0.95) + (white * 0.1);
            output[i] = lastOut * 0.8 + (Math.random() > 0.98 ? white * 0.15 : 0);
        } else if (type === 'cafe') {
            // Low rumble chatter simulation
            lastOut = (lastOut * 0.98) + (white * 0.05);
            const clink = Math.sin(i * 0.005) * (Math.random() > 0.9995 ? 0.2 : 0);
            output[i] = (lastOut * 1.5) + clink;
        }
    }
    return noiseBuffer;
}

function setSound(type) {
    try {
        initAudio();
        if (audioCtx.state === 'suspended') audioCtx.resume();
        
        // Stop current sound loop if active
        if (noiseNode) {
            noiseNode.stop();
            noiseNode.disconnect();
            noiseNode = null;
        }
        if (filterNode) {
            filterNode.disconnect();
            filterNode = null;
        }

        // Clean UI highlights
Use code with caution.

document.querySelectorAll('.sound-btn').forEach(b => b.classList.remove('active'));
document.getElementById(sound-${type}).classList.add('active');
if (type === 'none') return;
// Generate synthetic looping sounds safely without external files
noiseNode = audioCtx.createBufferSource();
noiseNode.buffer = generateNoiseBuffer(type);
noiseNode.loop = true;
filterNode = audioCtx.createBiquadFilter();
if (type === 'rain') {
filterNode.type = 'lowpass';
filterNode.frequency.value = 1200;
} else if (type === 'cafe') {
filterNode.type = 'bandpass';
filterNode.frequency.value = 400;
filterNode.Q.value = 0.6;
} else {
filterNode.type = 'lowpass';
filterNode.frequency.value = 8000;
}
const gainNode = audioCtx.createGain();
gainNode.gain.value = 0.5;
noiseNode.connect(filterNode);
filterNode.connect(gainNode);
gainNode.connect(audioCtx.destination);
noiseNode.start();
} catch (e) {
console.log("Audio not interacted yet", e);
}
}
